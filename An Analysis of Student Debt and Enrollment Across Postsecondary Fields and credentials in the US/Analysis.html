Data Cleaning

reg_data <- data %>%
select(MEDIAN_DEBT = DEBT_ALL_STGP_ANY_MDN,
       PROGRAM = CIPDESC,
       CRED_LEVEL = CREDLEV,
MEAN_DEBT = DEBT_ALL_STGP_ANY_MEAN,
COUNT = DEBT_ALL_STGP_ANY_N) %>%
  filter(!is.na(MEDIAN_DEBT),!is.na(CRED_LEVEL),!is.na(MEAN_DEBT))
